            \Require inst
            \Require s
            \For{$i=1 \to inst.n$}
                \State $min.l[i] = i$
            \EndFor
            \State $min.taille\_l = inst.n$
            \State $s.tour\_geant[0] = 0$
            \State $min.i = 0$
            \For{$i=1 \to inst.n$}
                \State $s.tour_geant[i] = which_min(inst.min)$
                \State $min.i = s.tour\_geant[i]$
            \EndFor
            \State $s.tour\_geant[inst.n+1]=0$
        \end{algorithmic}
